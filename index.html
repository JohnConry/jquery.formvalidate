// Validate form #1 on form submission
$('#form_1').submit(function(e) {
	e.preventDefault();
	$(this).formvalidate({
		onSuccess: function(O) {
			alert('Form #1 is valid!');
		}
	});
});

// Validate form #2 when the submit button is clicked
$('#submit_2').click(function(e) {
	e.preventDefault();

	// From the anchor element find the closest form element
	$(this).closest('form').formvalidate({
		// Let's modiyfy a validation message and add a custom validation method
		validations: {
			// Modify error message
			required: {
				text: '{0} is required (and error messages can be overwritten!).'
			},
			// Add a custom validation
			middle_name: {
				text: 'Your {0} cannot be "{2}" because this is a custom validation method and I said so.',
				func: function(input, params) {
					return input.toLowerCase() === params[0].toLowerCase() ? false : true;
				}
			}
		}
	});
});
